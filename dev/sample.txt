        // Setup mini calendar
        function setupCalendar() {
            const today = new Date();
            const currentMonth = document.getElementById('currentMonth');
            const calendarGrid = document.querySelector('.calendar-grid');
            
            // Set initial month
            updateCalendar(today.getFullYear(), today.getMonth());
            
            // Previous month button
            document.getElementById('prevMonth').addEventListener('click', () => {
                today.setMonth(today.getMonth() - 1);
                updateCalendar(today.getFullYear(), today.getMonth());
            });
            
            // Next month button
            document.getElementById('nextMonth').addEventListener('click', () => {
                today.setMonth(today.getMonth() + 1);
                updateCalendar(today.getFullYear(), today.getMonth());
            });
        }

        // Update calendar display
        function updateCalendar(year, month) {
            const currentMonth = document.getElementById('currentMonth');
            const calendarGrid = document.querySelector('.calendar-grid');
            
            // Update month display
            const monthNames = ["January", "February", "March", "April", "May", "June",
                "July", "August", "September", "October", "November", "December"];
            currentMonth.textContent = `${monthNames[month]} ${year}`;
            
            // Get first day of month and total days
            const firstDay = new Date(year, month, 1).getDay();
            const daysInMonth = new Date(year, month + 1, 0).getDate();
            const today = new Date();
            
            // Clear existing dates (skip first 7 days of week)
            for (let i = 7; i < calendarGrid.children.length; i++) {
                calendarGrid.removeChild(calendarGrid.children[i]);
            }
            
            // Add empty cells for days before first day of month
            for (let i = 0; i < firstDay; i++) {
                const emptyCell = document.createElement('div');
                emptyCell.className = 'calendar-date';
                calendarGrid.appendChild(emptyCell);
            }
            
            // Add days of month
            for (let day = 1; day <= daysInMonth; day++) {
                const dateCell = document.createElement('div');
                dateCell.className = 'calendar-date';
                dateCell.textContent = day;
                
                // Check if this date has events
                const dateStr = `${year}-${String(month + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
                const hasEvent = upcomingEvents.some(event => event.date === dateStr);
                
                // Mark today
                if (year === today.getFullYear() && month === today.getMonth() && day === today.getDate()) {
                    dateCell.classList.add('today');
                }
                
                // Mark days with events
                if (hasEvent) {
                    dateCell.classList.add('event-day');
                }
                
                calendarGrid.appendChild(dateCell);
            }
        }

        // Apply filters
        function applyFilters() {
            const eventType = document.getElementById('eventTypeFilter').value;
            const timeFrame = document.getElementById('timeFrameFilter').value;
            const status = document.getElementById('statusFilter').value;
            const searchTerm = document.getElementById('searchInput').value.toLowerCase();
            
            // Get active tab

            let events = activeTab === 'upcoming' ? upcomingEvents : previousEvents;
            
            let filtered = events;
            
            // Filter by event type

                filtered = filtered.filter(event => event.type === eventType);

                filtered = filtered.filter(event => event.status === status);

            // Filter by search term

                filtered = filtered.filter(event => 
                    event.title.toLowerCase().includes(searchTerm) ||
                    event.client.toLowerCase().includes(searchTerm) ||
                    event.location.toLowerCase().includes(searchTerm) ||
                    event.description.toLowerCase().includes(searchTerm)
                );
            }
            
            // Filter by time frame for upcoming events
            if (activeTab === 'upcoming' && timeFrame !== 'all') {
                const today = new Date();
                filtered = filtered.filter(event => {
                    const eventDate = new Date(event.date);
                    
                    switch(timeFrame) {
                        case 'today':
                            return eventDate.toDateString() === today.toDateString();
                        case 'week':
                            const nextWeek = new Date(today);
                            nextWeek.setDate(today.getDate() + 7);
                            return eventDate >= today && eventDate <= nextWeek;
                        case 'month':
                            const nextMonth = new Date(today);
                            nextMonth.setMonth(today.getMonth() + 1);
                            return eventDate >= today && eventDate <= nextMonth;
                        case 'next-month':
                            const startNextMonth = new Date(today.getFullYear(), today.getMonth() + 1, 1);
                            const endNextMonth = new Date(today.getFullYear(), today.getMonth() + 2, 0);
                            return eventDate >= startNextMonth && eventDate <= endNextMonth;
                        default:
                            return true;
                    }
                });
            }
            
            // Update UI based on filtered results
            updateEventsDisplay(activeTab, filtered);
        }

        // Update events display
        function updateEventsDisplay(tab, filteredEvents) {
            const activeView = document.querySelector('.view-btn.active').getAttribute('data-view');
            
            // Show empty state if no events
            if (filteredEvents.length === 0) {
                emptyState.style.display = 'block';
                if (tab === 'upcoming') {
                    upcomingGrid.style.display = 'none';
                    upcomingList.style.display = 'none';
                    upcomingTimeline.style.display = 'none';
                } else {
                    previousGrid.style.display = 'none';
                    previousList.style.display = 'none';
                }
                return;
            }
            
            emptyState.style.display = 'none';
            
            // Render filtered events
            if (tab === 'upcoming') {
                // Grid View
                upcomingGrid.innerHTML = filteredEvents.map(event => createEventCard(event)).join('');
                upcomingGrid.style.display = activeView === 'grid' ? 'grid' : 'none';
                
                // List View
                upcomingList.innerHTML = filteredEvents.map(event => createEventListItem(event)).join('');
                upcomingList.style.display = activeView === 'list' ? 'flex' : 'none';
                
                // Timeline View
                upcomingTimeline.innerHTML = filteredEvents.map(event => createTimelineItem(event)).join('');
                upcomingTimeline.style.display = activeView === 'timeline' ? 'block' : 'none';
            } else {
                // Grid View
                previousGrid.innerHTML = filteredEvents.map(event => createEventCard(event)).join('');
                previousGrid.style.display = activeView === 'grid' ? 'grid' : 'none';
                
                // List View
                previousList.innerHTML = filteredEvents.map(event => createEventListItem(event)).join('');
                previousList.style.display = activeView === 'list' ? 'flex' : 'none';
            }
        }
