CREATE OR REPLACE VIEW v_provider_full_profile AS
SELECT
    sp.service_id,

    -- Business name (fname + lname)
    sp.businessName AS business_name,

    -- Provider Profiles fields
    pp.intro,
    pp.background_image,
    pp.profile_pic,
    pp.background_text,
    pp.serviceType,

FROM service_providers sp
LEFT JOIN provider_profiles pp
    ON sp.service_id = pp.service_id;


CREATE OR REPLACE VIEW v_packages_with_provider AS
SELECT 
    p.package_id,
    p.title,
    p.details,
    p.price,
    p.bg_image_name,
    p.notes,

    sp.service_id,
    CONCAT(sp.fname, ' ', sp.lname) AS provider_name,
    sp.businessName,
    sp.serviceType,
    sp.district,

    pro.profile_pic,

    ROUND(AVG(r.rating), 1) AS avg_rating,
    COUNT(r.review_id) AS total_reviews

FROM packages p

JOIN service_providers sp
    ON p.service_id = sp.service_id

LEFT JOIN provider_profiles pro
    ON p.service_id = pro.service_id

LEFT JOIN reviews r
    ON sp.service_id = r.service_id

GROUP BY 
    p.package_id,
    sp.service_id,
    pro.profile_pic;

DELIMITER $$

CREATE TRIGGER after_review_insert
AFTER INSERT ON reviews
FOR EACH ROW
BEGIN
    UPDATE provider_ratings
    SET 
        total_reviews = total_reviews + 1,
        avg_rating = (
            (avg_rating * (total_reviews - 1) + NEW.rating) / total_reviews
        )
    WHERE service_id = NEW.service_id;
END$$

DELIMITER $$

CREATE TRIGGER after_like_insert
AFTER INSERT ON event_likes
FOR EACH ROW
BEGIN
    IF NEW.status = 'ON' THEN
        UPDATE event_posts
        SET like_count = like_count + 1
        WHERE event_id = NEW.event_id;
    END IF;
END$$

DELIMITER ;

DELIMITER $$

CREATE TRIGGER after_like_update
AFTER UPDATE ON event_likes
FOR EACH ROW
BEGIN
    IF OLD.status = 'ON' AND NEW.status = 'OFF' THEN
        UPDATE event_posts
        SET like_count = like_count - 1
        WHERE event_id = NEW.event_id;
    END IF;
END$$

DELIMITER ;

  `last_message_time` datetime DEFAULT NULL,

  PRIMARY KEY (`conversation_id`),
  UNIQUE KEY `client_id` (`client_id`,`provider_id`),
  KEY `provider_id` (`provider_id`)
) ENGINE=InnoDB AUTO_INCREMENT=37 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

AFTER INSERT ON messages

CREATE OR REPLACE VIEW v_conversations_with_provider AS
SELECT
    c.conversation_id,
    c.client_id,
    c.provider_id,
    c.last_message,
    c.last_message_time,
    c.created_at,
    c.updated_at,

    -- Provider full name
    CONCAT(sp.fname, ' ', sp.lname) AS provider_name,

    -- Provider profile picture
    pp.profile_pic

FROM conversations c
JOIN service_providers sp
    ON c.provider_id = sp.service_id
LEFT JOIN provider_profiles pp
    ON sp.service_id = pp.service_id;

CREATE OR REPLACE VIEW v_conversations_with_client AS
SELECT
    c.conversation_id,
    c.client_id,
    c.provider_id,
    c.last_message,
    c.last_message_time,
    c.created_at,
    c.updated_at,

    -- client full name
    cl.name,

    -- client profile picture
    cl.profile_pic

FROM conversations c
JOIN clients cl
    ON c.client_id = c.client_id;

DELIMITER $$

CREATE TRIGGER trg_provider_complaint_replacement
AFTER UPDATE ON provider_complaints
FOR EACH ROW
BEGIN
    IF NEW.resolution_type = 'PROVIDER_REPLACED'
       AND OLD.resolution_type <> 'PROVIDER_REPLACED' THEN

        INSERT INTO service_provider_replacements (
            source_type,
            complaint_id,
            event_id,
            old_service_id,
            reason,
            assigned_ic_id
        )
        VALUES (
            'PROVIDER_COMPLAINT',
            NEW.complaint_id,
            NEW.event_id,
            NEW.service_id,
            NEW.description,
            NEW.assigned_ic_id
        );

    END IF;
END$$

DELIMITER ;

CREATE TABLE service_provider_replacements (
    replacement_id INT AUTO_INCREMENT PRIMARY KEY,

    -- Source of replacement
    source_type ENUM('CLIENT_COMPLAINT', 'PROVIDER_COMPLAINT') NOT NULL,
    complaint_id INT NOT NULL,

    -- Event & provider details
    event_id INT NOT NULL,
    old_service_id INT NOT NULL,
    new_service_id INT DEFAULT NULL, -- assigned later by IC

    -- Status of replacement process
    replacement_status ENUM(
        'PENDING',
        'SEARCHING',
        'ASSIGNED',
        'CONFIRMED',
        'FAILED'
    ) DEFAULT 'PENDING',

    -- Notes
    reason TEXT DEFAULT NULL,

    -- IC handling
    assigned_ic_id INT DEFAULT NULL,

    -- Timestamps
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    -- Foreign Keys
    CONSTRAINT fk_spr_event
        FOREIGN KEY (event_id)
        REFERENCES events(event_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE,

    CONSTRAINT fk_spr_old_provider
        FOREIGN KEY (old_service_id)
        REFERENCES service_providers(service_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE,

    CONSTRAINT fk_spr_new_provider
        FOREIGN KEY (new_service_id)
        REFERENCES service_providers(service_id)
        ON DELETE SET NULL
        ON UPDATE CASCADE

) ENGINE=InnoDB;
